      import React, { useState } from 'react';
import TwitterWidget from "../TwitterWidget/TwitterWidget"
import Card from 'react-bootstrap/Card';
import Nav from 'react-bootstrap/Nav';

const TwitterTabs = () => {
    const twitterAccounts = [{id:1, account: 'BungieHelp'},{id:2, account: 'DestinyTheGame'},{id:3, account: 'Destiny2Team'}]
    //      {twitterAccounts.map(item => ( <li key={item.id}>{item.account}</li>))}

    const [activeTab, setTab] = useState('BungieHelp'); 
    const handleTabSelect = (tab) => {
      setTab(tab)
    }
  
    // Renders a timer based on selected tab
    const renderSelectedAccount = () => {
        for (let i = 0; i < twitterAccounts.length; i++) {
            if (activeTab === twitterAccounts[i].account){
                return <TwitterWidget screenName={twitterAccounts[i].account}/>
            }
        }
    }
    return(
      <Card>
      <Card.Header className='bg-dark'>
        <Nav variant="tabs">
            {twitterAccounts.map(item => ( 
                <Nav.Item>
                    <Nav.Link key={item.id} style={{color:'#b48608'}} active={activeTab === item.account} onClick={() => handleTabSelect(item.account)}>{item.account}</Nav.Link>
                </Nav.Item>
            ))}
        </Nav>
      </Card.Header>
      <Card.Body>
          {renderSelectedAccount()}
      </Card.Body>
    </Card>
    );
};

export default TwitterTabs
